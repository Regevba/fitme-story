# Design System v2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a complete, pixel-perfect, token-driven design system for FitTracker — three-tier architecture (Primitive → Semantic → Component), token pipeline (Figma Variables → JSON → Swift), strict 4pt spacing grid, motion tokens, WCAG AA contrast, Dynamic Type, and 6 new SwiftUI components.

**Architecture:** AppPalette.swift (private primitives) feeds AppTheme.swift (public semantics) feeds AppDesignSystemComponents.swift (components). A Style Dictionary pipeline generates DesignTokens.swift from design-tokens/tokens.json so CI can detect Figma↔code drift. Dark mode is deferred to Phase B8.

**Tech Stack:** SwiftUI 5, Swift 5.9, Xcode 16, Node.js 20+ (Style Dictionary 3.x), Make

```js
const StyleDictionary = require('style-dictionary');

// Custom format: outputs a Swift enum DesignTokens with nested namespaces
StyleDictionary.registerFormat({
  name: 'swift/enum',
  formatter: function({ dictionary }) {
    const header = [
      '// FitTracker/DesignSystem/DesignTokens.swift',
      '// AUTO-GENERATED — do not edit by hand.',
      '// Run `make tokens` to regenerate from design-tokens/tokens.json',
      `// Generated: ${new Date().toISOString().split('T')[0]}`,
      '',
      'public enum DesignTokens {',
    ].join('\n');

    // Group tokens by first path component, then second
    const groups = {};
    dictionary.allTokens.forEach(token => {
      const [cat, sub, name] = token.path;
      if (!groups[cat]) groups[cat] = {};
      if (!groups[cat][sub]) groups[cat][sub] = [];
      groups[cat][sub].push({ name, value: token.value });
    });

    const lines = [header];
    for (const [cat, subs] of Object.entries(groups)) {
      const catName = cat.charAt(0).toUpperCase() + cat.slice(1);
      lines.push(`    public enum ${catName} {`);
      for (const [sub, tokens] of Object.entries(subs)) {
        const subName = sub.charAt(0).toUpperCase() + sub.slice(1);
        lines.push(`        public enum ${subName} {`);
        for (const { name, value } of tokens) {
          const type = typeof value === 'number' ? 'Double' : 'String';
          const formatted = type === 'Double' ? value : `"${value}"`;
          lines.push(`            public static let ${name}: ${type} = ${formatted}`);
        }
        lines.push(`        }`);
      }
      lines.push(`    }`);
    }
    lines.push('}');
    return lines.join('\n') + '\n';
  }
});

module.exports = {
  source: ['design-tokens/tokens.json'],
  platforms: {
    swift: {
      prefix: '',
      files: [{
        destination: 'FitTracker/DesignSystem/DesignTokens.swift',
        format: 'swift/enum',
      }]
    }
  }
};
```

```swift
// MARK: - DEBUG contrast validator
// Logs any text/background pairings that fail WCAG AA (4.5:1) at launch.
// Only compiled in DEBUG builds — zero runtime cost in production.
#if DEBUG
enum ColorContrastValidator {
    struct Pair {
        let name: String
        let foreground: Double  // relative luminance
        let background: Double  // relative luminance
    }

    // Relative luminance of a linear sRGB channel value
    private static func linearise(_ c: Double) -> Double {
        c <= 0.04045 ? c / 12.92 : pow((c + 0.055) / 1.055, 2.4)
    }

    // Relative luminance of an rgba(0,0,0,alpha) colour on white background
    private static func luminanceBlackOnWhite(opacity: Double) -> Double {
        let ch = 1.0 - opacity   // composited channel on white
        return 0.2126 * linearise(ch) + 0.7152 * linearise(ch) + 0.0722 * linearise(ch)
    }

    private static func contrastRatio(fgLum: Double, bgLum: Double) -> Double {
        let l1 = max(fgLum, bgLum), l2 = min(fgLum, bgLum)
        return (l1 + 0.05) / (l2 + 0.05)
    }

    /// Call from App.init() or SceneDelegate in DEBUG builds.
    static func validate() {
        // Text on white background pairings
        let opacityPairs: [(name: String, opacity: Double)] = [
            ("Text.primary",   0.84),
            ("Text.secondary", 0.62),
            ("Text.tertiary",  0.55),
        ]
        for pair in opacityPairs {
            let fg = luminanceBlackOnWhite(opacity: pair.opacity)
            let bg = 1.0   // white
            let ratio = contrastRatio(fgLum: fg, bgLum: bg)
            if ratio < 4.5 {
                print("⚠️ WCAG FAIL [\(pair.name)] contrast \(String(format: "%.1f", ratio)):1 (need 4.5:1)")
            } else {
                print("✓ WCAG [\(pair.name)] \(String(format: "%.1f", ratio)):1")
            }
        }
    }
}
#endif
```

```swift
// FitTracker/DesignSystem/AppMotion.swift
// Motion design tokens.
// All animations in the app use these values — never inline literals.
import SwiftUI

enum AppMotion {
    enum Duration {
        static let fast:        Double = 0.15
        static let standard:    Double = 0.22
        static let slow:        Double = 0.38
        static let pageTransit: Double = 0.30
    }

    enum Easing {
        static let enter       = Animation.easeOut(duration: Duration.standard)
        static let exit        = Animation.easeIn(duration: Duration.fast)
        static let interactive = Animation.spring(response: 0.28, dampingFraction: 0.72)
        static let pageEnter   = Animation.easeOut(duration: Duration.pageTransit)
    }
}

// View modifier: applies animation only when reduce motion is off
struct ReducedMotionAnimation: ViewModifier {
    @Environment(\.accessibilityReduceMotion) private var reduceMotion
    let animation: Animation

    func body(content: Content) -> some View {
        content.animation(reduceMotion ? .none : animation, value: UUID())
    }
}

extension View {
    /// Applies `animation` unless the user has enabled Reduce Motion.
    func appMotion(_ animation: Animation) -> some View {
        modifier(ReducedMotionAnimation(animation: animation))
    }
}
```

```swift
struct AppCard<Content: View>: View {
    let tone: AppCardTone
    let density: AppCardDensity
    var accessibilityLabel: String? = nil
    @ViewBuilder let content: () -> Content

    init(
        tone: AppCardTone = .standard,
        density: AppCardDensity = .default,
        accessibilityLabel: String? = nil,
        @ViewBuilder content: @escaping () -> Content
    ) {
        self.tone = tone
        self.density = density
        self.accessibilityLabel = accessibilityLabel
        self.content = content
    }

    var body: some View {
        content()
            .padding(density.padding)
            .frame(maxWidth: .infinity, alignment: .leading)
            .background(background, in: RoundedRectangle(cornerRadius: AppRadius.large, style: .continuous))
            .overlay(
                RoundedRectangle(cornerRadius: AppRadius.large, style: .continuous)
                    .stroke(borderColor, lineWidth: 1)
            )
            .shadow(color: shadowColor, radius: AppShadow.cardRadius, y: AppShadow.cardYOffset)
            .accessibilityLabel(accessibilityLabel ?? "")
    }

    private var background: Color {
        switch tone {
        case .standard: return AppColor.Surface.materialLight
        case .elevated: return AppColor.Surface.elevated
        case .quiet:    return AppColor.Surface.primary
        case .inverse:  return AppColor.Surface.inverse
        }
    }

    private var borderColor: Color {
        switch tone {
        case .inverse:          return Color.white.opacity(0.08)
        case .elevated:         return AppColor.Border.subtle
        case .standard, .quiet: return AppColor.Border.strong.opacity(0.35)
        }
    }

    private var shadowColor: Color {
        tone == .quiet ? .clear : AppShadow.cardColor
    }
```

```swift
// MARK: - ViewModifiers

/// Applies the eyebrow label style: caption, semibold, uppercase, tracked.
struct AppEyebrowStyle: ViewModifier {
    var color: Color = AppColor.Text.secondary

    func body(content: Content) -> some View {
        content
            .font(AppText.captionStrong)
            .textCase(.uppercase)
            .tracking(1)
            .foregroundStyle(color)
    }
}

/// Applies a full text role: font + default foreground color.
enum AppTextRole {
    case primary, secondary, tertiary, inversePrimary, inverseSecondary, accent

    var color: Color {
        switch self {
        case .primary:         return AppColor.Text.primary
        case .secondary:       return AppColor.Text.secondary
        case .tertiary:        return AppColor.Text.tertiary
        case .inversePrimary:  return AppColor.Text.inversePrimary
        case .inverseSecondary: return AppColor.Text.inverseSecondary
        case .accent:          return AppColor.Accent.primary
        }
    }
}

struct AppTextRoleModifier: ViewModifier {
    let role: AppTextRole

    func body(content: Content) -> some View {
        content.foregroundStyle(role.color)
    }
}

extension View {
    /// Sets eyebrow label styling: uppercase, semibold, letter-spaced.
    func appEyebrowStyle(color: Color = AppColor.Text.secondary) -> some View {
        modifier(AppEyebrowStyle(color: color))
    }

    /// Sets foreground color based on a semantic text role.
    func appTextRole(_ role: AppTextRole) -> some View {
        modifier(AppTextRoleModifier(role: role))
    }
}
```
